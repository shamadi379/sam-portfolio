Zero Trust with Intune

5-step guide to implementing Zero Trust device compliance

Oct 7, 2025 5 min read

Zero Trust means "never trust, always verify." With Intune, you can enforce device compliance before granting access to corporate resources. Here's how to set it up quickly and effectively.

1 Create Device Compliance Policy 

# Create Windows 10/11 Compliance Policy
Connect-MSGraph

$compliancePolicy = @{
    '@odata.type' = '#microsoft.graph.windows10CompliancePolicy'
    displayName = 'Zero Trust - Windows Compliance'
    description = 'Enforce Zero Trust principles for Windows devices'
    passwordRequired = $true
    passwordMinimumLength = 8
    passwordRequiredType = 'alphanumeric'
    osMinimumVersion = '10.0.19041'  # Windows 10 2004
    defenderEnabled = $true
    defenderVersion = 'notConfigured'
    firewallEnabled = $true
    antivirusRequired = $true
    antiSpywareRequired = $true
    deviceThreatProtectionEnabled = $true
    deviceThreatProtectionRequiredSecurityLevel = 'medium'
}

New-DeviceCompliancePolicy -bodyParameter $compliancePolicy

2 Configure Conditional Access 

# Conditional Access: Require Compliant Devices
$caPolicy = @{
    displayName = 'Zero Trust - Require Compliant Device'
    state = 'enabled'
    conditions = @{
        applications = @{
            includeApplications = @('All')
            excludeApplications = @()
        }
        users = @{
            includeUsers = @('All')
            excludeUsers = @('admin@company.com')  # Emergency access
        }
        platforms = @{
            includePlatforms = @('windows', 'iOS', 'android')
        }
    }
    grantControls = @{
        operator = 'AND'
        builtInControls = @('compliantDevice', 'mfa')
    }
}

New-MgIdentityConditionalAccessPolicy -BodyParameter $caPolicy

3 Mobile App Protection 

# iOS App Protection Policy
$appProtection = @{
    '@odata.type' = '#microsoft.graph.iosManagedAppProtection'
    displayName = 'Zero Trust - iOS Protection'
    description = 'Protect corporate data on iOS devices'
    periodOfflineBeforeAccessCheck = 'PT12H'  # 12 hours
    periodOnlineBeforeAccessCheck = 'PT1H'    # 1 hour
    allowedInboundDataTransferSources = 'managedApps'
    allowedOutboundDataTransferDestinations = 'managedApps'
    allowedOutboundClipboardSharingLevel = 'managedAppsWithPasteIn'
    dataBackupBlocked = $true
    deviceComplianceRequired = $true
    managedBrowserToOpenLinksRequired = $true
    saveAsBlocked = $true
    periodOfflineBeforeWipeIsEnforced = 'P30D'  # 30 days
    pinRequired = $true
    maximumPinRetries = 5
    simplePinBlocked = $true
    minimumPinLength = 6
    fingerprintBlocked = $false
    apps = @(
        @{ 'mobileAppIdentifier' = @{ '@odata.type' = '#microsoft.graph.iosMobileAppIdentifier'; bundleId = 'com.microsoft.msedge' } }
        @{ 'mobileAppIdentifier' = @{ '@odata.type' = '#microsoft.graph.iosMobileAppIdentifier'; bundleId = 'com.microsoft.Office.Outlook' } }
    )
}

New-MgDeviceAppManagementIosManagedAppProtection -BodyParameter $appProtection

4 Device Configuration Profile 

# Windows Security Configuration
$deviceConfig = @{
    '@odata.type' = '#microsoft.graph.windows10GeneralConfiguration'
    displayName = 'Zero Trust - Windows Security'
    description = 'Enhanced security settings for Zero Trust'
    
    # BitLocker
    bitLockerSystemDrivePolicy = @{
        encryptionMethod = 'aesCbc256'
        startupAuthenticationRequired = $true
        startupAuthenticationTpmUsage = 'required'
        recoveryPasswordRotation = 'enabledForAzureAd'
    }
    
    # Windows Defender
    defenderCloudBlockLevel = 'high'
    defenderCloudExtendedTimeout = 50
    defenderDaysBeforeDeletingQuarantinedMalware = 30
    defenderDetectedMalwareActions = @{
        lowSeverity = 'quarantine'
        moderateSeverity = 'quarantine'
        highSeverity = 'remove'
        severeSeverity = 'remove'
    }
    
    # Network settings
    wifiBlockAutomaticConnectHotspots = $true
    bluetoothBlockAdvertising = $true
    bluetoothBlockPromiscuousMode = $true
    
    # Privacy
    privacyBlockInputPersonalization = $true
    privacyBlockPublishUserActivities = $true
}

New-MgDeviceManagementDeviceConfiguration -BodyParameter $deviceConfig

5 Monitor Compliance 

# Check Device Compliance Status
$complianceReport = Get-MgDeviceManagementManagedDevice | ForEach-Object {
    [PSCustomObject]@{
        DeviceName = $_.deviceName
        UserName = $_.userDisplayName
        OS = $_.operatingSystem
        ComplianceState = $_.complianceState
        LastSyncDateTime = $_.lastSyncDateTime
        ThreatState = $_.deviceThreatProtectionState
    }
}

# Group by compliance status
$compliance = $complianceReport | Group-Object ComplianceState
$compliance | Select-Object Name, Count

# Export non-compliant devices
$nonCompliant = $complianceReport | Where-Object {$_.ComplianceState -ne 'compliant'}
$nonCompliant | Export-Csv -Path "NonCompliantDevices.csv" -NoTypeInformation

Write-Output "Total devices: $($complianceReport.Count)"
Write-Output "Compliant: $(($complianceReport | Where-Object {$_.ComplianceState -eq 'compliant'}).Count)"
Write-Output "Non-compliant: $($nonCompliant.Count)"
