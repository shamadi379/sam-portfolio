1 Right-size Virtual Machines

Most VMs are oversized by 40-60%. Use performance data to right-size your instances and save immediately.

# Azure VM rightsizing analysis
$subscriptionId = "your-subscription-id"
$resourceGroup = "your-rg"

# Get VM utilization data (last 30 days)
$vms = Get-AzVM -ResourceGroupName $resourceGroup

foreach ($vm in $vms) {
    # Get CPU metrics
    $cpuMetrics = Get-AzMetric -ResourceId $vm.Id -MetricName "Percentage CPU" `
        -StartTime (Get-Date).AddDays(-30) -TimeGrain 01:00:00 -Statistic Average
    
    $avgCpu = ($cpuMetrics.Data.Average | Measure-Object -Average).Average
    
    # Get memory metrics (requires VM insights)
    $memMetrics = Get-AzMetric -ResourceId $vm.Id -MetricName "Available Memory Bytes" `
        -StartTime (Get-Date).AddDays(-30) -TimeGrain 01:00:00 -Statistic Average
    
    Write-Output "VM: $($vm.Name)"
    Write-Output "  Current Size: $($vm.HardwareProfile.VmSize)"
    Write-Output "  Avg CPU: $([math]::Round($avgCpu, 2))%"
    
    # Recommend downsizing if CPU < 20%
    if ($avgCpu -lt 20) {
        Write-Output "  üí° RECOMMENDATION: Consider downsizing to save ~30%" -ForegroundColor Yellow
    }
}

# Get cost savings estimate
Get-AzConsumptionUsageDetail | Where-Object {$_.InstanceName -like "*VM*"} | 
    Group-Object InstanceName | Select-Object Name, @{N="MonthlyCost";E={($_.Group.PretaxCost | Measure-Object -Sum).Sum}}

üí∞ Impact: Right-sizing typically saves 20-40% on compute costs with zero performance impact!

For predictable workloads running 24/7, reserved instances offer up to 72% savings compared to pay-as-you-go pricing.

# Analyze workloads for Reserved Instance opportunities
# Get VM usage patterns over last 90 days

$vms = Get-AzVM
$riCandidates = @()

foreach ($vm in $vms) {
    # Check if VM has been running consistently
    $powerStates = Get-AzVM -ResourceGroupName $vm.ResourceGroupName -Name $vm.Name -Status
    
    if ($powerStates.Statuses[-1].Code -eq "PowerState/running") {
        # Get historical usage
        $usageData = Get-AzMetric -ResourceId $vm.Id -MetricName "Percentage CPU" `
            -StartTime (Get-Date).AddDays(-90) -TimeGrain 01:00:00
        
        $uptime = ($usageData.Data | Where-Object {$_.Average -gt 0}).Count
        $totalHours = ($usageData.Data).Count
        $uptimePercentage = ($uptime / $totalHours) * 100
        
        if ($uptimePercentage -gt 80) {
            $riCandidates += [PSCustomObject]@{
                VMName = $vm.Name
                Size = $vm.HardwareProfile.VmSize
                Region = $vm.Location
                UptimePercentage = [math]::Round($uptimePercentage, 2)
                EstimatedMonthlySavings = "Calculate based on current spend"
            }
        }
    }
}

$riCandidates | Format-Table -AutoSize
Write-Output "Found $($riCandidates.Count) VMs suitable for Reserved Instances"

3 Implement Auto-scaling

Auto-scaling automatically adjusts resources based on demand, ensuring you only pay for what you need.

# Configure VM Scale Set with auto-scaling
$resourceGroup = "prod-rg"
$scaleSetName = "web-vmss"

# Create auto-scale profile
$profile = New-AzAutoscaleProfile -Name "WebTier-Profile" `
    -DefaultCapacity 2 -MaximumCapacity 10 -MinimumCapacity 1

# Scale out rule (CPU > 70%)
$scaleOutRule = New-AzAutoscaleRule -Name "ScaleOut" `
    -MetricName "Percentage CPU" -MetricResourceId "/subscriptions/.../resourceGroups/$resourceGroup/providers/Microsoft.Compute/virtualMachineScaleSets/$scaleSetName" `
    -Threshold 70 -TimeAggregation Average -TimeGrain 00:01:00 -TimeWindow 00:05:00 `
    -ScaleActionDirection Increase -ScaleActionType ChangeCount -ScaleActionValue 1 `
    -ScaleActionCooldown 00:05:00

# Scale in rule (CPU < 30%)
$scaleInRule = New-AzAutoscaleRule -Name "ScaleIn" `
    -MetricName "Percentage CPU" -MetricResourceId "/subscriptions/.../resourceGroups/$resourceGroup/providers/Microsoft.Compute/virtualMachineScaleSets/$scaleSetName" `
    -Threshold 30 -TimeAggregation Average -TimeGrain 00:01:00 -TimeWindow 00:10:00 `
    -ScaleActionDirection Decrease -ScaleActionType ChangeCount -ScaleActionValue 1 `
    -ScaleActionCooldown 00:10:00

# Add rules to profile
$profile.Rules.Add($scaleOutRule)
$profile.Rules.Add($scaleInRule)

# Create autoscale setting
New-AzAutoscaleSetting -Name "WebTier-AutoScale" -ResourceGroupName $resourceGroup `
    -TargetResourceId "/subscriptions/.../resourceGroups/$resourceGroup/providers/Microsoft.Compute/virtualMachineScaleSets/$scaleSetName" `
    -Profile $profile -Enabled $true

‚è∞ Best Practice: Set longer cool-down periods for scale-in to avoid thrashing and unnecessary costs!

4 Clean Up Unused Resources

Identify and remove orphaned resources that continue to generate costs without providing value.

# Comprehensive cost cleanup script
$subscriptionId = "your-subscription-id"
$report = @()

# Find unattached disks
$unattachedDisks = Get-AzDisk | Where-Object {$_.DiskState -eq "Unattached"}
foreach ($disk in $unattachedDisks) {
    $cost = ($disk.DiskSizeGB * 0.05) # Rough estimate: $0.05/GB/month
    $report += [PSCustomObject]@{
        Resource = "Disk: $($disk.Name)"
        Type = "Unattached Disk"
        EstimatedMonthlyCost = "`$$cost"
        Action = "Delete if not needed"
    }
}

# Find unused NICs
$unusedNICs = Get-AzNetworkInterface | Where-Object {$_.VirtualMachine -eq $null}
foreach ($nic in $unusedNICs) {
    $report += [PSCustomObject]@{
        Resource = "NIC: $($nic.Name)"
        Type = "Unused Network Interface"
        EstimatedMonthlyCost = "`$5"
        Action = "Delete if orphaned"
    }
}

# Find stopped VMs (still charged for storage)
$stoppedVMs = Get-AzVM -Status | Where-Object {$_.PowerState -eq "VM deallocated"}
foreach ($vm in $stoppedVMs) {
    $report += [PSCustomObject]@{
        Resource = "VM: $($vm.Name)"
        Type = "Deallocated VM"
        EstimatedMonthlyCost = "Storage costs still apply"
        Action = "Delete if no longer needed"
    }
}

# Find empty resource groups
$emptyRGs = Get-AzResourceGroup | Where-Object {
    (Get-AzResource -ResourceGroupName $_.ResourceGroupName).Count -eq 0
}
foreach ($rg in $emptyRGs) {
    $report += [PSCustomObject]@{
        Resource = "RG: $($rg.ResourceGroupName)"
        Type = "Empty Resource Group"
        EstimatedMonthlyCost = "`$0 (but cleanup recommended)"
        Action = "Delete if not needed"
    }
}

$report | Format-Table -AutoSize
$totalDisks = $unattachedDisks.Count
$totalNICs = $unusedNICs.Count
Write-Output "`nSummary: Found $totalDisks unused disks and $totalNICs unused NICs"

5 Optimize Storage Costs

Storage often represents 30-40% of cloud costs. Right-tier your storage and implement lifecycle policies for massive savings.

# Storage cost optimization
$storageAccounts = Get-AzStorageAccount

foreach ($account in $storageAccounts) {
    $ctx = $account.Context
    
    # Analyze blob storage usage
    $containers = Get-AzStorageContainer -Context $ctx
    
    foreach ($container in $containers) {
        $blobs = Get-AzStorageBlob -Container $container.Name -Context $ctx
        
        # Find blobs older than 90 days
        $oldBlobs = $blobs | Where-Object {$_.LastModified -lt (Get-Date).AddDays(-90)}
        
        # Estimate savings by moving to Cool tier
        $hotStorage = $blobs | Where-Object {$_.Properties.AccessTier -eq "Hot"}
        $coolSavings = ($hotStorage | Measure-Object -Property Length -Sum).Sum * 0.01 / 1GB
        
        Write-Output "Container: $($container.Name)"
        Write-Output "  Total blobs: $($blobs.Count)"
        Write-Output "  Old blobs (>90 days): $($oldBlobs.Count)"
        Write-Output "  Hot tier savings potential: `$$([math]::Round($coolSavings, 2))/month"
    }
    
    # Set lifecycle policy for automatic tiering
    $rule = @{
        enabled = $true
        name = "MoveToCoolgstafter30days"
        type = "Lifecycle"
        definition = @{
            filters = @{
                blobTypes = @("blockBlob")
            }
            actions = @{
                baseBlob = @{
                    tierToCool = @{
                        daysAfterModificationGreaterThan = 30
                    }
                    tierToArchive = @{
                        daysAfterModificationGreaterThan = 90
                    }
                }
            }
        }
    }
    
    # Apply lifecycle policy
    Set-AzStorageAccountManagementPolicy -ResourceGroupName $account.ResourceGroupName `
        -StorageAccountName $account.StorageAccountName -Policy @{rules=@($rule)}
}

Set Up Cost Alerts 

# Create budget alerts for cost monitoring
$subscriptionId = "your-subscription-id"
$budgetName = "Monthly-Budget-Alert"
$budgetAmount = 5000  # Set your monthly budget

# Create budget
$budget = @{
    name = $budgetName
    type = "Microsoft.Consumption/budgets"
    properties = @{
        category = "Cost"
        amount = $budgetAmount
        timeGrain = "Monthly"
        timePeriod = @{
            startDate = (Get-Date -Format "yyyy-MM-01")
        }
        notifications = @{
            "Actual_GreaterThan_80_Percent" = @{
                enabled = $true
                operator = "GreaterThan"
                threshold = 80
                contactEmails = @("admin@company.com")
                contactRoles = @("Owner", "Contributor")
                contactGroups = @()
            }
            "Forecast_GreaterThan_100_Percent" = @{
                enabled = $true
                operator = "GreaterThan"
                threshold = 100
                contactEmails = @("admin@company.com")
                contactRoles = @("Owner")
                contactGroups = @()
            }
        }
    }
}

# Deploy budget using REST API
$uri = "https://management.azure.com/subscriptions/$subscriptionId/providers/Microsoft.Consumption/budgets/$budgetName" + "?api-version=2019-10-01"
$headers = @{
    'Authorization' = "Bearer $((Get-AzAccessToken).Token)"
    'Content-Type' = 'application/json'
}

Invoke-RestMethod -Uri $uri -Method PUT -Body ($budget | ConvertTo-Json -Depth 5) -Headers $headers
